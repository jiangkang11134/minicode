# === dict_utils.py ===
"""字典工具模块。"""

from typing import Any, Dict, List


def find_keys_by_value(d: Dict, target: Any) -> List:
    """找到字典中所有值为 target 的键。

    Args:
        d: 输入字典。
        target: 要查找的目标值。

    Returns:
        所有值为 target 的键组成的列表；若没有匹配键，返回空列表。
    """
    return [k for k, v in d.items() if v == target]


def find_all_values_by_keys(d: Dict, keys: List) -> List:
    """根据键列表从字典中取出所有存在的值。

    Args:
        d: 输入字典。
        keys: 要取值的键列表。

    Returns:
        所有存在的键对应的值组成的列表；不存在的键被跳过。
    """
    return [d[k] for k in keys if k in d]

# === demo.py ===
"""dict_utils 使用演示。"""

from dict_utils import find_all_values_by_keys, find_keys_by_value


def demo_find_keys_by_value():
    """演示 find_keys_by_value：值 -> 键查询。"""
    print("=" * 50)
    print("1. find_keys_by_value(d, target)")
    print("=" * 50)

    d = {"apple": "fruit", "carrot": "vegetable", "banana": "fruit"}
    target = "fruit"
    result = find_keys_by_value(d, target)
    print(f"   字典: {d}")
    print(f"   查询值: {target!r}")
    print(f"   结果键: {result}\n")


def demo_find_all_values_by_keys():
    """演示 find_all_values_by_keys：键列表 -> 值列表。"""
    print("=" * 50)
    print("2. find_all_values_by_keys(d, keys)")
    print("=" * 50)

    scores = {"Alice": 95, "Bob": 87, "Charlie": 92, "Diana": 78}
    names = ["Alice", "Charlie", "Eve"]
    result = find_all_values_by_keys(scores, names)
    print(f"   字典: {scores}")
    print(f"   查询键: {names}")
    print(f"   结果值: {result}  (Eve 不存在，被跳过)\n")
